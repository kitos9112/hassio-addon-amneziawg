#include <tunables/global>

# AppArmor profile for the AmneziaWG Server add-on.
#
# Confines the writable filesystem to /data, /config, /tmp and /run and scopes
# exec to the expected paths. Capabilities/networking are left broad on purpose:
# they are already gated by the add-on's declared `privileged: [NET_ADMIN]` and
# the container's Docker capability set, and narrowing them further (e.g. to
# `capability net_admin,` or dropping `mount`) risks breaking s6-overlay init.
# That tightening is a TODO pending validation on a real Home Assistant OS box.
#
# If the add-on fails to start with AppArmor "DENIED" lines, please open an issue
# with the denied line so the profile can be corrected. Disabling the profile
# (`apparmor: false`) removes this confinement entirely and is a last resort.
profile amneziawg flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities (restricted to the container cap set by Docker regardless).
  capability,

  # Networking: TUN, netlink (ip/awg), raw + UDP sockets, masquerade.
  network,

  file,
  signal,
  mount,
  umount,
  remount,

  # S6-Overlay init + service supervision
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /sbin/** ix,
  /usr/sbin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/s6-overlay/** rix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,

  # Bashio + add-on libraries
  /usr/lib/bashio/** ix,
  /usr/lib/amneziawg/** r,

  # Userspace WireGuard control + runtime state
  /dev/net/tun rw,
  /run/{,**} rwk,
  /var/run/{,**} rwk,
  /tmp/{,**} rwk,
  /dev/tty rw,

  # Persistent data (keys, configs) + client export share
  /data/{,**} rw,
  /config/{,**} rw,

  # Forwarding toggle + interface/route introspection
  @{PROC}/sys/net/ipv4/** rw,
  @{PROC}/** r,
  /sys/** r,
}
